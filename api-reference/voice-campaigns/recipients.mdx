---
title: "Get Voice Campaign Recipients"
api: "GET /voice-campaigns/{uuid}/recipients"
description: "Retrieve the list of recipients for a specific voice campaign"
---

<ParamField path="uuid" type="string" required>
  The unique identifier (UUID) of the campaign
</ParamField>

      "duration": 30
    },
    {
      "id": 2,

      "status": "no_answer",
      "called_at": "2026-01-01T10:06:00.000000Z",
